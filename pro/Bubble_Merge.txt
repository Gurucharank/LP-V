#include <iostream>
#include <omp.h>
using namespace std;

void bubble(int a[],int n){
    for(int k=0;k<n;k++){
        #pragma omp parallel for
        for(int i=k%2;i<n-1;i+=2)
            if(a[i]>a[i+1]) swap(a[i],a[i+1]);
    }
}

void merge(int a[],int l,int m,int r){
    int t[1000],i=l,j=m+1,k=0;
    while(i<=m&&j<=r) t[k++]=a[i]<a[j]?a[i++]:a[j++];
    while(i<=m) t[k++]=a[i++];
    while(j<=r) t[k++]=a[j++];
    for(i=l,k=0;i<=r;i++) a[i]=t[k++];
}

void mergesort(int a[],int l,int r){
    if(l>=r) return;
    int m=(l+r)/2;

    #pragma omp parallel sections
    {
        #pragma omp section
        mergesort(a,l,m);

        #pragma omp section
        mergesort(a,m+1,r);
    }

    merge(a,l,m,r);
}

int main(){
    int a[]={9,5,2,7,1,4},n=6;

    bubble(a,n);
    cout<<"Bubble Sort: ";
    for(int i:a) cout<<i<<" ";

    int b[]={9,5,2,7,1,4};
    mergesort(b,0,n-1);

    cout<<"\nMerge Sort: ";
    for(int i:b) cout<<i<<" ";
}
